## 六、附录：实际代码片段

### mergeCells 实现 (JSA880.js:7902)

```javascript
RngUtils.z合并单元格 = function(rng, strategy, options) {
    if (!isWPS) return null;
    var r = typeof rng === 'string' ? Range(rng) : rng;
    if (strategy === undefined) {
        r.Merge();
        return r;
    }
    strategy = strategy || 'cm';
    options = options || {};
    switch (strategy) {
        case 'cm': this._mergeColumnLevel(r, options); break;
        case 'c': this._mergeByColumn(r); break;
        case 'r': this._mergeByRow(r); break;
        case 'rm': this._mergeRowThenColumn(r); break;
        default: Console.log('RngUtils.mergeCells: 未知策略 "' + strategy + '"');
    }
    return r;
};
```

### unMergeCells 实现 (JSA880.js:8006)

```javascript
RngUtils.z取消合并单元格 = function(rng, fillAllRows) {
    if (!isWPS) return null;
    var r = typeof rng === 'string' ? Range(rng) : rng;
    if (fillAllRows === undefined) fillAllRows = true;
    for (var i = 1; i <= r.Rows.Count; i++) {
        for (var j = 1; j <= r.Columns.Count; j++) {
            var cell = r.Cells(i, j);
            if (cell.MergeCells && cell.MergeArea.Count > 1) {
                var mergeArea = cell.MergeArea;
                var val = mergeArea.Value2;
                mergeArea.UnMerge();
                if (fillAllRows) {
                    mergeArea.Value2 = val;
                }
            }
        }
    }
};
```
